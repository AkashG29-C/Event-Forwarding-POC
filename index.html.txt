    function getDeviceInfo() {
      return {
        screenHeight: window.screen.height,
        screenWidth: window.screen.width,
        screenOrientation: (screen.orientation || {}).type || "unknown"
      };
    }

    // Wait for Alloy to be available
    function waitForAlloy(callback) {
      if (typeof alloy === "function") {
        callback();
      } else {
        setTimeout(function () {
          waitForAlloy(callback);
        }, 100);
      }
    }

    waitForAlloy(function () {
      // Optional: Re-configure alloy manually if needed
      alloy("configure", {
        edgeConfigId: "993a5934-4f41-4827-a059-00508caf6325",
        orgId: "0CEB60F754C7E06B0A4C98A2@AdobeOrg",
        debugEnabled: true
      });

      // Pageview event
      document.addEventListener("DOMContentLoaded", function () {
        console.log("Page loaded â€“ sending pageview event");

        alloy("sendEvent", {
          xdm: {
            eventType: "pageView",
            timestamp: new Date().toISOString(),
            web: {
              webPageDetails: {
                name: "Home Page",
                URL: window.location.href
              }
            },
            device: getDeviceInfo()
          }
        });
      });
    });
