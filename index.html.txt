  <script>
    function waitForAlloy(callback) {
      if (typeof alloy === "function") {
        callback();
      } else {
        setTimeout(() => waitForAlloy(callback), 100);
      }
    }

    function getDeviceInfo() {
      return {
        screenHeight: window.screen.height,
        screenWidth: window.screen.width,
        screenOrientation: screen.orientation?.type || "unknown"
      };
    }

    waitForAlloy(function () {
      alloy("configure", {
        edgeConfigId: "993a5934-4f41-4827-a059-00508caf6325",
        orgId: "0CEB60F754C7E06B0A4C98A2@AdobeOrg",
        debugEnabled: true
      });

      // Page View Event
      document.addEventListener("DOMContentLoaded", function () {
        alloy("sendEvent", {
          xdm: {
            eventType: "pageView",
            web: {
              webPageDetails: {
                name: "Home Page",
                URL: window.location.href
              },
              webReferrer: {
                URL: document.referrer
              }
            },
            device: getDeviceInfo()
          }
        });
      });

      // Email Submit Button (now via class)
      document.querySelector(".emailSubmitBtn").addEventListener("click", function (e) {
        e.preventDefault();
        const email = document.getElementById("emailInput").value;
        alloy("sendEvent", {
          xdm: {
            eventType: "formSubmit",
            identityMap: {
              email: [{ id: email, authenticatedState: "ambiguous", primary: true }]
            },
            device: getDeviceInfo()
          }
        });
      });

      // Name Submit
      document.querySelector(".nameSubmitBtn").addEventListener("click", function (e) {
        e.preventDefault();
        const name = document.getElementById("nameInput").value;
        alloy("sendEvent", {
          xdm: {
            eventType: "formSubmit",
            identifier: name,
            _accenture_partner: {
              BizBanking: true
            },
            device: getDeviceInfo()
          }
        });
      });

      // Age Submit
      document.querySelector(".ageSubmitBtn").addEventListener("click", function (e) {
        e.preventDefault();
        const age = document.getElementById("ageInput").value;
        alloy("sendEvent", {
          xdm: {
            eventType: "formSubmit",
            identifier: age,
            _accenture_partner: {
              BizBanking: false
            },
            device: getDeviceInfo()
          }
        });
      });
    });
  </script>
